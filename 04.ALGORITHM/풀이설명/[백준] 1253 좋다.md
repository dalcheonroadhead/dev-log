import java.io.*;
import java.util.*;

public class Main {

    /*
    * b1253 μΆ‹λ‹¤
    * (1) Target κ°’ μ°ΎκΈ° O(N)
    * (2) Target κ°’ λ§λ“¤ μ μλ”μ§€ ν™•μΈ O(N)

    public static void main(String[] args) throws IOException {
        // μ…λ ¥ λ°›κΈ°
        int ans = 0;
        int N;
        int [] arr;
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        N = Integer.parseInt(br.readLine());
        arr = new int [N];
        StringTokenizer st = new StringTokenizer(br.readLine());
        for (int i = 0; i < N; i++) {
            arr[i] = Integer.parseInt(st.nextToken());
        }
        // κ³„μ‚°
        Arrays.sort(arr);
        for (int i = 0; i < N; i++) {
            int target = arr[i];
            int l = 0; int r = N-1;
            while (l < r){
                if(l == i){
                    l++;
                    continue;
                }else if(r == i){
                    r--;
                    continue;
                }
                int now = arr[l] + arr[r];
                if(now < target) l++;
                else if(now > target) r--;
                else {
                    ans++;
                    break;
                }
            }
        }
        System.out.println(ans);
    }
}
